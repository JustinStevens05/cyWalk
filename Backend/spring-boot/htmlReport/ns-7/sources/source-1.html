 @Entity
 public class StepGoal {
     @Id

     private Long id;

     private int weeklyGoal;
 
     @OneToOne

     }

     public int getWeeklyGoal() {
         return weeklyGoal;
     }
 
     public void setWeeklyGoal(int weeklyGoal) {
         this.weeklyGoal = weeklyGoal;
     }

         return people;
     }

         this.people = people;
     }
